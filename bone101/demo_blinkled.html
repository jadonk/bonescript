Bonescript webpage app template

This is a minimal example that can be extended easily.

led = bone.P8_13;
state = 0;

pinMode(led, 'out');
toggleLED = function() {
 state = state ? 0 : 1;
 digitalWrite(led, state);
};

timer = setInterval(toggleLED, 100);

stopTimer = function() {
 clearInterval(timer);
};

setTimeout(stopTimer, 3000);
